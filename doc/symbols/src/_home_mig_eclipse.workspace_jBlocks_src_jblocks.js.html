 12 /**
 13  * <p>jBlocks is a javascript library that can be used as a jQuery plugin and/or a framework taking care of the layout in a HTML5 application. 
 14  * It allows designing applications that will run on desktops as well as on mobile devices without having to worry much about the details 
 15  * of screen resolution, pixel density and so on.</p>
 16  * 
 17  * <p>The basic idea is to rely on a javascript-run algorithm to perform the layout rather than regular CSS like float or table layouts. 
 18  * jBlocks arranges rectangular blocks into a viewport according to a set of loose preferences given by the developer. The library tries to 
 19  * minimize the spare areas without content in real time. This means the application may add, remove or modify blocks geometry and jBlocks 
 20  * will recalculate and adjust the layout accordingly.</p>
 21  * <p>The resulting layout method is not really deterministic so it may not be suitable to all applications: please do not use jBlocks to run a 
 22  * nuclear plant !</p>
 23  */

 40 	  // Create a new Class that inherits from this class
 41 	  Class.extend = function(prop) {
 42 	    var _super = this.prototype;
 43 	   
 44 	    // Instantiate a base class (but only create the instance,
 45 	    // don't run the init constructor)
 46 	    initializing = true;
 47 	    var prototype = new this();
 48 	    initializing = false;
 49 	   
 50 	    // Copy the properties over onto the new prototype
 51 	    for (var name in prop) {
 52 	      // Check if we're overwriting an existing function
 53 	      prototype[name] = typeof prop[name] == "function" &&
 54 	        typeof _super[name] == "function" && fnTest.test(prop[name]) ?
 55 	        (function(name, fn){
 56 	          return function() {
 57 	            var tmp = this._super;
 58 	           
 59 	            // Add a new ._super() method that is the same method
 60 	            // but on the super-class
 61 	            this._super = _super[name];
 62 	           
 63 	            // The method only need to be bound temporarily, so we
 64 	            // remove it when we're done executing
 65 	            var ret = fn.apply(this, arguments);       
 66 	            this._super = tmp;
 67 	           
 68 	            return ret;
 69 	          };
 70 	        })(name, prop[name]) :
 71 	        prop[name];
 72 	    }

 76 	      // All construction is actually done in the init method

 88 	    Class.prototype.constructor = Class;

134 	 * <p>A jBlocks context defines a rectangular viewport in which smaller sized blocks are layouted. Depending on the the setup, vertical or 
135 	 * horizontal scrolling is managed. If no scrolling is defined, some blocks may not be shown if they canâ€™t fit into the space.</p>
136 	 * 
137 	 * <p>jBlocks can be used recursively by defining blocks that are themselves jBlocks contexts handling their own layout. This is convenient 
138 	 * to define structured blocks containing several zones, like for instance a header, content area and footer.</p>
139 	 * 
140 	 * <p>A viewport has a virtual size, for instance 18x12. jBlocks adapts the real pixel size to the device physical geometry. This virtual 
141 	 * size can be controlled by the application but jBlocks provides a default behavior to adjust to the actual device. For instance, on a 
142 	 * full size big desktop screen, the size might be 30x24, causing many blocks to be seen, while when run on a mobile phone, it can

248 	JBlocks.prototype.getWorker=function() {
249 		var $this=this;
250 		if(typeof Worker=="undefined") {
251 			return null;
252 		}
253 		if(this.worker)
254 			return this.worker;
255 		if(this.options.ownWorker)
256 			this.worker = CreateWorker(this.options.workerURL);
257 		else {
258 			if(!globalWorker)
259 				globalWorker = CreateWorker(this.options.workerURL);
260 			this.worker = globalWorker;
261 		}
262 		$(document).bind("jblocks.worker", function(event,data) {
263 			if(data.jBlocksIndex!=$this.jBlocksIndex)
264 				return;
265 			switch(data.cmd) {
266 			case "layouts":
267 				//debugger;
268 				$this.layoutInProgress--;
269 				$this.layoutCache[data.args.signature]={ 
270 					layouts: JSON.parse(JSON.stringify(data.args.layouts)), 
271 					layoutData: JSON.parse(JSON.stringify(data.args.layoutData)),
272 					lastUsed: new Date().getTime(),
273 				};
274 				$this.checkSanity(data.args.layouts,data.args.layoutData);
275 				$this.layouts=data.args.layouts;
276 				$this.layoutData=data.args.layoutData;
277 				for(var i in data.args.layouts) {
278 					var layout=data.args.layouts[i];
279 					var block=$this.blocks[layout.block.id];
280 					if(block && block.options.earlyLayout==false)
281 						$this.applyBlockLayout(layout);
282 				}
283 				if(data.args.signature==$this.layoutSign) {
284 					$this.layoutDone();
285 				}
286 				break;
287 			case "layout":
288 				var blockId=data.args.layout.block.id;
289 				if(data.args.misfit)
290 					$this.layoutData.misfits[blockId]=true;
291 				else
292 					delete $this.layoutData.misfits[blockId];
293 				var block=$this.blocks[blockId];
294 				if(block) {
295 					if(block.options.earlyLayout)
296 						$this.applyBlockLayout(data.args.layout);
297 				}
298 				break;
299 			}
300 		});
301 		return this.worker;
302 	}

538 	JBlocks.prototype.applyBlockLayout=function(layout) {
539 		var $this=this;
540 		var block=this.blocks[layout.block.id];
541 		if(typeof block=="undefined") {
542 			$.jBlocks("log","Unknown block id",layout.block.id);
543 			this.anchor.find("#jblocks-block-"+this.jBlocksIndex+"-"+layout.block.id).hide();
544 		} else {
545 			block.actualLayout=layout;
546 			if(block.layout.visible==false || this.layoutData.misfits[block.id]) {
547 				block.anchor.hide();
548 			} else {
549 				block.anchor.show();
550 				var position=block.anchor.position();
551 				var oldGeometry={
552 					width: block.anchor.width(),
553 					height: block.anchor.height(),
554 					top: position.top,
555 					left: position.left,
556 				}
557 				var newGeometry={
558 					width: Math.round(layout.w*this.geometry.uWidth),
559 					height: Math.round(layout.h*this.geometry.uHeight),
560 					left: Math.round(layout.c*this.geometry.uWidth),
561 					top: Math.round(layout.r*this.geometry.uHeight),
562 				}
563 				var modifiedSize=oldGeometry.width!=newGeometry.width || oldGeometry.height!=newGeometry.height;
564 				if(modifiedSize) {
565 					var contentGeo=block.getContentGeometry(newGeometry,$this.geometry.uWidth,$this.geometry.uHeight);
566 					block.onStartResize(contentGeo,newGeometry);
567 					(function(block,newGeometry) {
568 						contentGeo.top=newGeometry.top;
569 						contentGeo.left=newGeometry.left;
570 						var randColor="rgba(230,230,230,0.5)";
571 						block.anchor.stop().animate(contentGeo,block.options.resizeAnimateDelay,function() {
572 							block.content.show();
573 							block.onResize(newGeometry);					
574 						});
575 					})(block,newGeometry);
576 				} else if(oldGeometry.left!=newGeometry.left || oldGeometry.top!=newGeometry.top) {
577 					block.anchor.stop().animate({
578 						left: Math.round(layout.c*this.geometry.uWidth),
579 						top: Math.round(layout.r*this.geometry.uHeight),
580 					},block.options.resizeAnimateDelay,function() {
